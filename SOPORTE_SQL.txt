-- EJECUTAR ESTOS SCRIPTS EN EL SQL EDITOR DE SUPABASE PARA ACTIVAR LAS NUEVAS FUNCIONALIDADES

-- 1. TABLA PARA DOCUMENTOS DE EVENTOS (Presupuestos, Imágenes, etc.)
CREATE TABLE IF NOT EXISTS event_documents (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  event_id UUID REFERENCES events(id) ON DELETE CASCADE,
  nombre_archivo TEXT NOT NULL,
  url TEXT NOT NULL,
  storage_path TEXT NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

ALTER TABLE event_documents ENABLE ROW LEVEL SECURITY;
DO $$ 
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_policies WHERE policyname = 'Acceso total para todos event_docs') THEN
    CREATE POLICY "Acceso total para todos event_docs" ON event_documents FOR ALL USING (true) WITH CHECK (true);
  END IF;
END $$;


-- 2. TABLA PARA TÍTULOS PERSONALIZADOS DE EMPLEADOS
CREATE TABLE IF NOT EXISTS employee_titles (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  profile_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  nombre TEXT NOT NULL,
  fecha_caducidad DATE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

ALTER TABLE employee_titles ENABLE ROW LEVEL SECURITY;
DO $$ 
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_policies WHERE policyname = 'Acceso total títulos personalizados') THEN
    CREATE POLICY "Acceso total títulos personalizados" ON employee_titles FOR ALL USING (true) WITH CHECK (true);
  END IF;
END $$;


-- 3. TABLA PARA EQUIPOS DE PROTECCIÓN INDIVIDUAL (EPIs)
CREATE TABLE IF NOT EXISTS employee_epis (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  profile_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  epi_nombre TEXT NOT NULL,
  modelo TEXT,
  referencia TEXT,
  norma_ce TEXT,
  proveedor TEXT,
  fecha_fabricacion DATE,
  fecha_caducidad DATE,
  fecha_compra DATE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

ALTER TABLE employee_epis ENABLE ROW LEVEL SECURITY;
DO $$ 
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_policies WHERE policyname = 'Acceso total epis') THEN
    CREATE POLICY "Acceso total epis" ON employee_epis FOR ALL USING (true) WITH CHECK (true);
  END IF;
END $$;
